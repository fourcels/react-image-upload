## Install

```bash npm2yarn
npm install react-image-upload
```

## Usage

```jsx
import { ImageUpload } from "react-image-upload";

function MyApp(props) {
  return <ImageUpload />;
}
```

### Basic

```jsx live
<ImageUpload />
```

### Advanced

```jsx live
<ImageUpload
  value={[
    "/img/undraw_docusaurus_mountain.svg",
    "/img/undraw_docusaurus_react.svg",
  ]}
  max={5}
  dropzoneOptions={{ onDragEnter: (evt) => console.log(evt) }}
  photoProviderProps={{ maskOpacity: 0.8 }}
  onChange={(value) => console.log(value)}
  onUpload={(file) =>
    new Promise((resolve) =>
      setTimeout(
        () => resolve(URL.createObjectURL(file)),
        1000 * Math.floor(Math.random() * 5)
      )
    )
  }
/>
```

### Readonly

```jsx live
<ImageUpload
  value={[
    "/img/undraw_docusaurus_mountain.svg",
    "/img/undraw_docusaurus_react.svg",
    "/img/undraw_docusaurus_tree.svg",
  ]}
  readonly
/>
```
