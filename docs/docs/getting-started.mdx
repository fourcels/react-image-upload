## Install

```bash npm2yarn
npm install @fourcels/react-image-upload
```

## Usage

```jsx
import { ImageUpload } from "@fourcels/react-image-upload";
import "@fourcels/react-image-upload/style.css";

function MyApp() {
  return <ImageUpload />;
}
```

### Basic

```jsx live
<ImageUpload />
```

### Advanced

```jsx live
<ImageUpload
  value={[
    "/react-image-upload/img/undraw_docusaurus_mountain.svg",
    "/react-image-upload/img/undraw_docusaurus_react.svg",
  ]}
  max={5}
  dropzoneOptions={{ onDragEnter: (evt) => console.log(evt) }}
  photoProviderProps={{ maskOpacity: 0.8 }}
  onChange={(value) => console.log(value)}
  onUpload={(file) =>
    new Promise((resolve) =>
      setTimeout(
        () => resolve(URL.createObjectURL(file)),
        1000 * Math.floor(Math.random() * 5)
      )
    )
  }
/>
```

### Custom

```jsx live
<ImageUpload>
  <button className="button button--secondary">Upload</button>
</ImageUpload>
```

### Preview

```jsx live
<ImagePreview
  value={[
    "/react-image-upload/img/undraw_docusaurus_mountain.svg",
    "/react-image-upload/img/undraw_docusaurus_react.svg",
    "/react-image-upload/img/undraw_docusaurus_tree.svg",
  ]}
/>
```

### Preview - Placeholder

```jsx live
<ImagePreview placeholder="/react-image-upload/img/undraw_docusaurus_mountain.svg" />
```

### Preview - Provider Placeholder

```jsx live
<ImageProvider placeholder="/react-image-upload/img/undraw_docusaurus_mountain.svg">
  <ImagePreview />
</ImageProvider>
```
